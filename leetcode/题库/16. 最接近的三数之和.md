# 16. 最接近的三数之和
## 关键点
1. 注意只是前缀。
## 解法
1. 先找到第一个和第二个的公共前缀，然后用这个公共前缀和第三个字符串比较，以此类推。
#### 代码(367ms)
```python
class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        min_value = inf
        ans = 0
        for i in range(n):
            r,l = i+1,n-1
            while r<l:
                t_sum = nums[i]+nums[r]+nums[l]
                t = abs(t_sum - target)
                if t < min_value:
                    ans = t_sum
                    min_value = t
                if t_sum > target:
                    l-=1
                elif t_sum < target:
                    r+=1
                else:
                    break
        return ans
```
#### 代码(3ms)
```python
class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        min_value = inf
        ans = 0
        for i in range(n-2):
            #如果当前值和前一个值一样，那直接进入下一个值
            if i > 0 and nums[i] == nums[i-1]:
                continue
            # 因为现在nums为递增序列，如果连着三个数组的和都大于target了，
            # 那么后面的肯定也大于，就不用再计算了
            s1 = nums[i] + nums[i+1] + nums[i+2]
            if s1 > target:
                if s1-target < min_value:
                    ans = s1
                break
            #如果当前值加上数组的最后两个值都小于target，那就直接下一个数值。
            s2 = nums[i] + nums[n-1] + nums[n-2]
            if s2 < target:
                if target - s2 < min_value:
                    ans = s2
                    min_value = target - s2
                continue
            r,l = i+1,n-1
            while r<l:
                t_sum = nums[i]+nums[r]+nums[l]
                t = abs(t_sum - target)
                if t < min_value:
                    ans = t_sum
                    min_value = t
                if t_sum > target:
                    l-=1
                elif t_sum < target:
                    r+=1
                else:
                    return ans
        return ans
```
