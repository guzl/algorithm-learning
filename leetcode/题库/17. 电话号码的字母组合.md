def letterCombinations(self, digits: str) -> List[str]:
    mydict = {  '2':'abc',
                '3':'def',
                '4':'ghi',
                '5':'jkl',
                '6':'mno',
                '7':'pqrs',
                '8':'tuv',
                '9':'wxyz'}
    ans = []
    for x in digits:
        newstr = []
        for y in mydict[x]:
            if len(ans) == 0:
                newstr = [i for i in mydict[x]]
                break
            else:
                newstr += [i+y for i in ans]
        ans = newstr
    return ans
