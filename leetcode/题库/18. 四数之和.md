# 18. 四数之和
## 关键点
1. 无。
## 解法
和三数之和一样，只是需要枚举两个数字，然后剩下的两个数字用左右指针遍历。
#### 代码(19ms)
```python
class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:
        nums.sort()
        n = len(nums)
        ans = []
        for i in range(n-3):
            #如果当前值和前一个值一样，那直接进入下一个值
            if i > 0 and nums[i] == nums[i-1]:
                continue
            if nums[i] + nums[i+1] + nums[i+2] + nums[i+3] > target:
                break
            if nums[i] + nums[-3] + nums[-2] + nums[-1] < target:
                continue
            for j in range(i+1,n-2):
                #如果当前值和前一个值一样，那直接进入下一个值
                if j > i+1 and nums[j] == nums[j-1]:
                    continue
                if nums[i] + nums[j] + nums[j+1] + nums[j+2] > target:
                    break
                if nums[i] + nums[j] + nums[-2] + nums[-1] < target:
                    continue
                r,l = j+1,n-1
                while r<l:
                    t_sum = nums[i]+nums[j]+nums[r]+nums[l]
                    if t_sum > target:
                        l-=1
                    elif t_sum < target:
                        r+=1
                    else:
                        ans.append([nums[i],nums[j],nums[r],nums[l]])
                        r+=1
                        while r<l and nums[r] == nums[r - 1]:
                            r +=1
                        l-=1
                        while r<l and nums[l] == nums[l + 1]:
                            l-=1
        return ans
```
